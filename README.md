Esta vista permite agregar y a su vez editar productos de la base de datos, dejando total manejo del cliente, lo cual da un valor agregado

# Códigos de las tablas (SQL)

```bash
-- Tabla pedidos
CREATE TABLE IF NOT EXISTS public.pedidos (
  id BIGINT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  fecha TIMESTAMPTZ NOT NULL DEFAULT now(),
  nombre_empresa TEXT NOT NULL,
  correo_empresa TEXT NOT NULL,
  observaciones TEXT,
  total NUMERIC NOT NULL
);

-- Tabla productos
CREATE TABLE IF NOT EXISTS public.productos (
  id INTEGER PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  nombre TEXT NOT NULL UNIQUE,
  precio_unitario NUMERIC NOT NULL
);

-- Tabla detalle_pedido (depende de pedidos)
CREATE TABLE IF NOT EXISTS public.detalle_pedido (
  id BIGINT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  pedido_id BIGINT NOT NULL REFERENCES public.pedidos(id),
  producto TEXT NOT NULL,
  cantidad NUMERIC NOT NULL,
  precio_unitario NUMERIC NOT NULL,
  subtotal NUMERIC NOT NULL
);
```
